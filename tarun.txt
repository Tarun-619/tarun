1) Remove duplicates from a sorted array


#include <stdio.h>

int removeDuplicates(int* nums, int numsSize) {
    if (numsSize == 0)
    return 0; 
    int cn = 1;

    for (int i = 1; i < numsSize; i++) {
        if (nums[i] != nums[cn - 1]) { 
            nums[cn] = nums[i]; 
            cn++; 
           
        } 
    
    }
    return cn;
}
